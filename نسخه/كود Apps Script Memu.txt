const ss = SpreadsheetApp.getActiveSpreadsheet();

// 1. دالة جلب البيانات (للموقع ولوحة التحكم)
function doGet(e) {
  const action = e.parameter.action;
  
  if (action === 'getMenu') {
    return getSheetData("Menu");
  }
  
  if (action === 'getOrders') {
    return getSheetData("Orders");
  }
}

// 2. دالة استقبال البيانات (من لوحة التحكم لإضافة أصناف)
function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const sheet = ss.getSheetByName(data.targetSheet); // بيبعت لـ Menu أو Orders
    
    // إضافة الصف الجديد بالبيانات اللي جاية من الكمبيوتر
    sheet.appendRow(data.rowData);
    
    return ContentService.createTextOutput(JSON.stringify({ "status": "success", "message": "تمت الإضافة بنجاح" }))
      .setMimeType(ContentService.MimeType.JSON);
      
  } catch (error) {
    return ContentService.createTextOutput(JSON.stringify({ "status": "error", "message": error.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}

// دالة مساعدة لتحويل بيانات الشيت لـ JSON
function getSheetData(sheetName) {
  const sheet = ss.getSheetByName(sheetName);
  const values = sheet.getDataRange().getValues();
  const headers = values[0];
  const rows = values.slice(1);
  
  const data = rows.map(row => {
    let obj = {};
    headers.forEach((header, i) => {
      obj[header] = row[i];
    });
    return obj;
  });
  
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
